CREATE TABLE IF NOT EXISTS category(id INT NOT NULL PRIMARY KEY, name varchar(50) NOT NULL, description varchar(200) NOT NULL);

CREATE TABLE IF NOT EXISTS productImage(product_id INT NOT NULL, url varchar(20) NOT NULL, description varchar(200) NOT NULL, FOREIGN KEY(product_id) REFERENCES products(id) ON DELETE CASCADE);
